17def _check_arrays(X, y=None): 

18 if isinstance(X, list): 

19 raise ValueError("X must be a numpy array") 

20 if not len(X.shape) == 2: 

21 raise ValueError("X must be a 2D array. Try X[:, numpy.newaxis]") 

22 try: 

23 if y is None: 

24 return 

25 except AttributeError: 

26 if not len(y.shape) == 1: 

27 raise ValueError("y must be a 1D array.") 

28 

29 if not len(y) == X.shape[0]: 

30 raise ValueError("X and y must contain the" "same number of samples")

33def no_information_rate(targets, predictions, loss_fn): 

34 combinations = np.array(list(product(targets, predictions))) 

35 return loss_fn(combinations[:, 0], combinations[:, 1]) 

36 

37 

38def accuracy(targets, predictions): 

39 return np.mean(np.array(targets) == np.array(predictions)) 

40 

41 

42def mse(targets, predictions): 

43 return np.mean((np.array(targets) - np.array(predictions)) ** 2)

46def bootstrap_point632_score( 

47 estimator, 

48 X, 

49 y, 

50 n_splits=200, 

51 method=".632", 

52 scoring_func=None, 

53 predict_proba=False, 

54 random_seed=None, 

55 clone_estimator=True, 

56 **params, 

57): 

58 """ 

59 Implementation of the .632 [1] and .632+ [2] bootstrap 

60 for supervised learning 

61 

62 References: 

63 

64 - [1] Efron, Bradley. 1983. "Estimating the Error Rate 

65 of a Prediction Rule: Improvement on Cross-Validation." 

66 Journal of the American Statistical Association 

67 78 (382): 316. doi:10.2307/2288636. 

68 - [2] Efron, Bradley, and Robert Tibshirani. 1997. 

69 "Improvements on Cross-Validation: The .632+ Bootstrap Method." 

70 Journal of the American Statistical Association 

71 92 (438): 548. doi:10.2307/2965703. 

72 

73 Parameters 

74 ---------- 

75 estimator : object 

76 An estimator for classification or regression that 

77 follows the scikit-learn API and implements "fit" and "predict" 

78 methods. 

79 

80 X : array-like 

81 The data to fit. Can be, for example a list, or an array at least 2d. 

82 

83 y : array-like, optional, default: None 

84 The target variable to try to predict in the case of 

85 supervised learning. 

86 

87 n_splits : int (default=200) 

88 Number of bootstrap iterations. 

89 Must be larger than 1. 

90 

91 method : str (default='.632') 

92 The bootstrap method, which can be either 

93 - 1) '.632' bootstrap (default) 

94 - 2) '.632+' bootstrap 

95 - 3) 'oob' (regular out-of-bag, no weighting) 

96 for comparison studies. 

97 

98 scoring_func : callable, 

99 Score function (or loss function) with signature 

100 ``scoring_func(y, y_pred, **kwargs)``. 

101 If none, uses classification accuracy if the 

102 estimator is a classifier and mean squared error 

103 if the estimator is a regressor. 

104 

105 predict_proba : bool 

106 Whether to use the `predict_proba` function for the 

107 `estimator` argument. This is to be used in conjunction 

108 with `scoring_func` which takes in probability values 

109 instead of actual predictions. 

110 For example, if the scoring_func is 

111 :meth:`sklearn.metrics.roc_auc_score`, then use 

112 `predict_proba=True`. 

113 Note that this requires `estimator` to have 

114 `predict_proba` method implemented. 

115 

116 random_seed : int (default=None) 

117 If int, random_seed is the seed used by 

118 the random number generator. 

119 

120 clone_estimator : bool (default=True) 

121 Clones the estimator if true, otherwise fits 

122 the original. 

123 

124 params : additional parameters 

125 Additional parameters to be passed to the .fit() function of the 

126 estimator when it is fit to the bootstrap samples. 

127 

128 

129 Returns 

130 ------- 

131 scores : array of float, shape=(len(list(n_splits)),) 

132 Array of scores of the estimator for each bootstrap 

133 replicate. 

134 

135 Examples 

136 -------- 

137 >>> from sklearn import datasets, linear_model 

138 >>> from mlxtend.evaluate import bootstrap_point632_score 

139 >>> iris = datasets.load_iris() 

140 >>> X = iris.data 

141 >>> y = iris.target 

142 >>> lr = linear_model.LogisticRegression() 

143 >>> scores = bootstrap_point632_score(lr, X, y) 

144 >>> acc = np.mean(scores) 

145 >>> print('Accuracy:', acc) 

146 0.953023146884 

147 >>> lower = np.percentile(scores, 2.5) 

148 >>> upper = np.percentile(scores, 97.5) 

149 >>> print('95%% Confidence interval: [%.2f, %.2f]' % (lower, upper)) 

150 95% Confidence interval: [0.90, 0.98] 

151 

152 For more usage examples, please see 

153 https://rasbt.github.io/mlxtend/user_guide/evaluate/bootstrap_point632_score/ 

154 

155 """ 

156 if not isinstance(n_splits, int) or n_splits < 1: 

157 raise ValueError( 

158 "Number of splits must be" " greater than 1. Got %s." % n_splits 

159 ) 

160 

161 allowed_methods = (".632", ".632+", "oob") 

162 if not isinstance(method, str) or method not in allowed_methods: 

163 raise ValueError( 

164 "The `method` must " "be in %s. Got %s." % (allowed_methods, method) 

165 ) 

166 

167 # Pandas compatibility 

168 if hasattr(X, "values"): 

169 X = X.values 

170 if hasattr(y, "values"): 

171 y = y.values 

172 

173 _check_arrays(X, y) 

174 

175 if clone_estimator: 

176 cloned_est = clone(estimator) 

177 else: 

178 cloned_est = estimator 

179 

180 if scoring_func is None: 

181 if cloned_est._estimator_type == "classifier": 

182 scoring_func = accuracy 

183 elif cloned_est._estimator_type == "regressor": 

184 scoring_func = mse 

185 else: 

186 raise AttributeError( 

187 "Estimator type undefined." "Please provide a scoring_func argument." 

188 ) 

189 

190 # determine which prediction function to use 

191 # either label, or probability prediction 

192 if not predict_proba: 

193 predict_func = cloned_est.predict 

194 else: 

195 if not getattr(cloned_est, "predict_proba", None): 

196 raise RuntimeError( 

197 f"The estimator {cloned_est} does not " 

198 f"support predicting probabilities via " 

199 f"`predict_proba` function." 

200 ) 

201 predict_func = cloned_est.predict_proba 

202 

203 oob = BootstrapOutOfBag(n_splits=n_splits, random_seed=random_seed) 

204 scores = np.empty(dtype=float, shape=(n_splits,)) 

205 cnt = 0 

206 

207 for train, test in oob.split(X): 

208 cloned_est.fit(X[train], y[train], **params) 

209 

210 # get the prediction probability 

211 # for binary class uses the last column 

212 predicted_test_val = predict_func(X[test]) 

213 

214 if method in (".632", ".632+"): 

215 # predictions on the internal training set: 

216 # predicted_train_val = predict_func(X[train]) 

217 

218 # compute training error on the whole training set as reported in 

219 # the original .632 boostrap paper 

220 # in Eq (6.12) in 

221 # "Estimating the Error Rate of a Prediction Rule: Improvement 

222 # on Cross-Validation" 

223 # by B. Efron, 1983, https://doi.org/10.2307/2288636 

224 # Also see the discussion at 

225 # https://github.com/rasbt/mlxtend/discussions/828 

226 # 

227 # This also applies to the .632+ estimate in the paper 

228 # "Improvements on Cross-Validation: The .632+ Bootstrap Method" 

229 # https://www.tandfonline.com/doi/abs/10.1080/01621459.1997.10474007 

230 predicted_train_val = predict_func(X) 

231 

232 if predict_proba: 

233 len_uniq = np.unique(y) 

234 

235 if len(len_uniq) == 2: 

236 predicted_train_val = predicted_train_val[:, 1] 

237 predicted_test_val = predicted_test_val[:, 1] 

238 

239 test_acc = scoring_func(y[test], predicted_test_val) 

240 

241 if method == "oob": 

242 acc = test_acc 

243 

244 else: 

245 test_err = 1 - test_acc 

246 

247 # training error on the whole training set as mentioned in the 

248 # previous comment above 

249 train_err = 1 - scoring_func(y, predicted_train_val) 

250 

251 if method == ".632+": 

252 gamma = 1 - ( 

253 no_information_rate(y, cloned_est.predict(X), scoring_func) 

254 ) 

255 R = (test_err - train_err) / (gamma - train_err) 

256 weight = 0.632 / (1 - 0.368 * R) 

257 

258 else: 

259 weight = 0.632 

260 

261 acc = 1 - (weight * test_err + (1.0 - weight) * train_err) 

262 

263 scores[cnt] = acc 

264 cnt += 1 

265 return scores
